Classes

    constructor(text) { this.text = text; }
    greet() { return "Hello " + this.text + "!"; }
}

Promises

function readFile(path) {
    return new Promise((resolve, reject) => {
        fs.readFile(path, (error, data) => {
            if (error) { reject(error); return; }
            resolve(data);
        });
    };
}

Promises in Angular 2: Services
